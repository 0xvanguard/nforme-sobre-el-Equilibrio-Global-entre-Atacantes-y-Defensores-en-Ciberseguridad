    historical_registry = load_historical_jobs()
    logging.info("Iniciando secuencia de escaneo agresivo de portales de empleo globales...")
    
    # Construcción de una consulta booleana altamente restrictiva para aislar perfiles de entrada

            search_term=search_query,
            location="Colombia", # Parámetro adaptable para búsquedas globales
            results_wanted=100,
            hours_old=24, # Filtro de frescura para garantizar ser el primer postulante
            country_indeed='Colombia',
            proxies=PROXY_LIST # Rotación de red para eludir el límite de tasa de los servidores
        )
        
        discovery_counter = 0
        for index, row in extracted_jobs_df.iterrows():
            # Generación de un hash único compuesto para identificar unívocamente la vacante
            unique_job_hash = f"{row['company']}_{row['title']}_{row['date_posted']}"
            
            # Evaluación de novedad y disparo condicional de alertas
            if unique_job_hash not in historical_registry:
                dispatch_tactical_alert(row)
                historical_registry.add(unique_job_hash)
                discovery_counter += 1
                
        persist_historical_jobs(historical_registry)
        logging.info(f"Ciclo operativo completado exitosamente. Se identificaron {discovery_counter} nuevas oportunidades de alto valor.")
        
    except Exception as pipeline_error:
        logging.error(f"Excepción fatal no controlada en el proceso de extracción de datos: {pipeline_error}")
